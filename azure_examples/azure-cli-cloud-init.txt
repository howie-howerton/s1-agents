# Use Azure CLI to Bootstrap a VM with the SentinelOne Agent

# Create a Resource Group
az group create --name myResourceGroupAutomate --location eastus

# Create a VM (referencing a cloud-init script 'azure-cloud-init.yaml - Be sure to substitute valid values for AGENT_VERSION and SITE_TOKEN)
az vm create \
    --resource-group myResourceGroupAutomate \
    --name myAutomatedVM \
    --image UbuntuLTS \
    --admin-username azureuser \
    --generate-ssh-keys \
    --custom-data azure-cloud-init.yaml

# Use the value from publicIpAddress to SSH to your VM with:  ssh azureuser@<publicIpAddress_here>

# Delete the VM when you're finished:  az vm delete --resource-group myResourceGroupAutomate --name myAutomatedVM -y
